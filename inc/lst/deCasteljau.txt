deCasteljau(P, n, u, C)
{
    /*Вычисление точки на кривой Безье*/
    /*[in]: P, n, u*/
    /*[out]: C (точка)*/
    for(i=0; i<=n; i++) /* Используем локальный массив, */
        Q[i] = P[i]     /* чтобы не изменить исходный массив контрольных точек */
    for(k=1; k<=n; k++)
        for(i=0; i<=n-k; i++)
            Q[i] = (1.0-u)*Q[i] + u*Q[i+1]
    C = Q[0]
}
